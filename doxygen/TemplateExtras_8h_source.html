22 // These two templates are like `AsmPrinter::{,detect_}has_print_method`,
23 // except they detect print methods taking `raw_ostream` (not `AsmPrinter`).
24 template <typename T>
25 using has_print_method =
26  decltype(std::declval<T>().print(std::declval<llvm::raw_ostream &>()));
27 template <typename T>
28 using detect_has_print_method = llvm::is_detected<has_print_method, T>;
29 template <typename T, typename R = void>
30 using enable_if_has_print_method =
31  std::enable_if_t<detect_has_print_method<T>::value, R>;
32 
33 /// Generic template for defining `operator<<` overloads which delegate
34 /// to `T::print(raw_ostream&) const`. Note that there's already another
35 /// generic template which defines `operator<<(AsmPrinterT&, T const&)`
36 /// via delegating to `operator<<(raw_ostream&, T const&)`.
37 template <typename T>
38 inline enable_if_has_print_method<T, llvm::raw_ostream &>
39 operator<<(llvm::raw_ostream &os, T const &t) {
40  t.print(os);
41  return os;
42 }

45 /// Convert an enum to its underlying type. This template is designed
46 /// to avoid introducing implicit conversions to other integral types,
47 /// and is a backport of C++23 `std::to_underlying`.
48 template <typename Enum>
49 constexpr std::underlying_type_t<Enum> to_underlying(Enum e) noexcept {
50  return static_cast<std::underlying_type_t<Enum>>(e);
51 }

54 template <typename T>
55 static constexpr bool IsZeroCostAbstraction =
56  // These two predicates license the compiler to make several optimizations;
57  // some of which are explicitly documented by the C++ standard:
58  // <https://en.cppreference.com/w/cpp/types/is_trivially_copyable#Notes>
59  // <https://en.cppreference.com/w/cpp/types/is_trivially_destructible#Notes>
60  // However, some key optimizations aren't mentioned by the standard; e.g.,
61  // that trivially-copyable enables passing-by-value, and the conjunction
62  // of trivially-copyable and trivially-destructible enables passing those
63  // values in registers rather than on the stack (cf.,
64  // <https://www.agner.org/optimize/calling_conventions.pdf>).
65  std::is_trivially_copyable_v<T> && std::is_trivially_destructible_v<T> &&
66  // This one helps ensure ABI compatibility (e.g., padding and alignment):
67  // <https://en.cppreference.com/w/cpp/types/is_standard_layout#Notes>
68  // <https://en.cppreference.com/w/cpp/language/classes#Standard-layout_class>
69  // In particular, the standard mentions that passing/returning a `struct`
70  // by value can sometimes introduce ABI overhead compared to using
71  // `enum class`; so this assertion is attempting to avoid that.
72  // <https://en.cppreference.com/w/cpp/language/enum#enum_relaxed_init_cpp17>
73  std::is_standard_layout_v<T> &&
74  // These two are what SmallVector uses to determine whether it can
75  // use memcpy. The commentary there mentions that it's intended to be
76  // an approximation of `is_trivially_copyable`, so this may be redundant
77  // with the above, but we include it just to make sure.
78  std::is_trivially_copy_constructible<T>::value &&
79  std::is_trivially_move_constructible<T>::value;
