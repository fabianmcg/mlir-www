36 /// Apply the given transformation \p toApply to every element of the tensor to
37 /// be transformed \p toTransform.
38 ///
39 /// Elements of \p toTransform are extracted as \p SrcValueType.
40 ///
41 /// \returns A tensor with the same size as \p toTransform, containing
42 /// \p TargetValueType values of type \p TargetType.
43 template <class SrcValType, class TargetValType, class TargetType>
44 DenseElementsAttr applyElementWise(
45  const DenseElementsAttr &toTransform,
46  const std::function<TargetValType(const SrcValType &, TargetType)> &toApply,
47  TargetType targetType) {
48  SmallVector<TargetValType> transformedValues;
49  // We already know the amount of values we will insert, reserve space for
50  // all of them to avoid dynamic resizing
51  transformedValues.reserve(toTransform.getNumElements());
52  for (auto val : toTransform.getValues<SrcValType>()) {
53  auto transformedVal = toApply(val, targetType);
54  transformedValues.push_back(transformedVal);
55  }
56 
57  // Make sure that the output tensor has the expected output type
58  auto inShape = toTransform.getType();
59  auto outTy = inShape.cloneWith({}, targetType);
60 
61  return DenseElementsAttr::get(outTy, transformedValues);
62 }

80 /// Function that checks if \p toCheck is a dense TOSA constant tensor.
81 LogicalResult notifyIfNoTosaDenseConstantTensor(TypedValue<TensorType> toCheck,
82  TosaOp location,
83  PatternRewriter &rewriter) {
84  // Check whether the tensor is constant and dense
85  // TODO We currently ensure the tensor is dense by using the correct type for
86  // the bind_value, however we do not actually need this value. It would be
87  // nicer to only have a check here.
88  DenseElementsAttr tmp;
89  if (!matchPattern(toCheck, m_Constant(&tmp))) {
90  return rewriter.notifyMatchFailure(location,
91  "Non-const or non-dense input tensor");
92  }
93 
94  // Make sure it actually is a TOSA constant (the match allows for other
95  // constants as well)
96  if (isa<ConstOp>(toCheck.getDefiningOp())) {
97  return success();
98  }
99 
100  return rewriter.notifyMatchFailure(location,
101  "The reciprocal can only be folded if "
102  "it operates on a TOSA constant");
103 }

105 /// Function that checks if \p toCheck is a dense TOSA constant float tensor.
106 LogicalResult notifyIfNotConstantFloatTosaTensor(TypedValue<TensorType> toCheck,
107  TosaOp location,

116 /// Heuristic to decide when to replace a unary operation on a constant with the
117 /// folded value.
118 /// Folding operations on constants can lead to an increased memory usage
119 /// whenever the input cannot be replaced but a new constant is inserted. Hence,
120 /// this will currently only suggest folding when the memory impact is
121 /// negligible.
122 /// Takes the \p unaryOp and the constant input \p values.
123 /// \returns Whether folding should be applied.
124 bool constantUnaryOpShouldBeFolded(TosaOp unaryOp, DenseElementsAttr values) {
125  assert(unaryOp->getNumOperands() == 1);
126  auto inputOp = unaryOp->getOperand(0);
127 
128  // If the input is a splat, we don't care for the number of users
129  if (isa<SplatElementsAttr>(values)) {
130  return true;
131  }
132 
133  // If this is the only use of the tensor it should be replaced as no
134  // additional memory is required
135  return inputOp.hasOneUse();
136 }

138 template <typename BaseType>
139 DenseElementsAttr transposeType(ElementsAttr attr, ShapedType inputType,
140  ShapedType outputType,
141  llvm::ArrayRef<int64_t> permValues) {
142  if (inputType.getNumElements() == 0)
143  return DenseElementsAttr::get(outputType, llvm::ArrayRef<BaseType>{});
144 
145  auto attrValues = attr.getValues<BaseType>();
146  auto inputShape = inputType.getShape();
147 
148  // The inverted permutation map and strides of the output are used to compute
149  // the contribution of a given dimension to the destination linear index in
150  // an order-independent way.
151  auto outputStrides = computeStrides(outputType.getShape());
152  auto invertedPermValues = invertPermutationVector(permValues);
153 
154  auto initialValue = *std::begin(attrValues);
155  SmallVector<BaseType> outputValues(inputType.getNumElements(), initialValue);
156 
157  for (const auto &it : llvm::enumerate(attrValues)) {
158  auto srcLinearIndex = it.index();
159 
160  uint64_t dstLinearIndex = 0;
161  for (int64_t dim = inputShape.size() - 1; dim >= 0; --dim) {
162  // Compute the index into the current dimension of the source vector.
163  auto sourceIndexForDim = srcLinearIndex % inputShape[dim];
164  srcLinearIndex /= inputShape[dim];
165 
166  // Add the contribution of the current dimension to the output using the
167  // permutation map.
168  dstLinearIndex +=
169  outputStrides[invertedPermValues[dim]] * sourceIndexForDim;
170  }
171 
172  outputValues[dstLinearIndex] = it.value();
173  }
174 
175  return DenseElementsAttr::get(outputType,
176  llvm::ArrayRef<BaseType>(outputValues));
177 }

179 // A type specialized transposition of an ElementsAttr.
180 // This implementation tries to operate on the underlying data in its raw
181 // representation when possible to avoid allocating a large number of Attribute
182 // objects.
183 DenseElementsAttr transpose(ElementsAttr attr, ShapedType inputType,
184  ShapedType outputType,
185  llvm::ArrayRef<int64_t> permValues) {
186  auto baseType = inputType.getElementType();
187 
188  // Handle possible integer types
189  if (auto intType = dyn_cast<IntegerType>(baseType)) {
190  switch (intType.getWidth()) {
191  case 1:
192  return transposeType<bool>(attr, inputType, outputType, permValues);
193  case 8:
194  return transposeType<int8_t>(attr, inputType, outputType, permValues);
195  case 16:
196  return transposeType<int16_t>(attr, inputType, outputType, permValues);
197  case 32:
198  return transposeType<int32_t>(attr, inputType, outputType, permValues);
199  case 64:
200  return transposeType<int64_t>(attr, inputType, outputType, permValues);
201  default:
202  return transposeType<APInt>(attr, inputType, outputType, permValues);
203  }
204  }
205 
206  // Handle possible float types
207  if (baseType.isF32()) {
208  return transposeType<float>(attr, inputType, outputType, permValues);
209  }
210 
211  return transposeType<APFloat>(attr, inputType, outputType, permValues);
212 }

214 struct TosaFoldConstantTranspose : public OpRewritePattern<tosa::TransposeOp> {
215  using OpRewritePattern::OpRewritePattern;
216 
217  LogicalResult matchAndRewrite(tosa::TransposeOp op,
218  PatternRewriter &rewriter) const override {
219  auto outputType = cast<ShapedType>(op.getType());
220  // TOSA supports quantized types.
221  if (!outputType.getElementType().isIntOrIndexOrFloat())
222  return failure();
223 
224  ElementsAttr inputValues;
225  if (!matchPattern(op.getInput1(), m_Constant(&inputValues)))
226  return failure();
227  // Make sure the input is a constant that has a single user.
228  if (!llvm::hasSingleElement(op.getInput1().getDefiningOp()->getUsers()))
229  return failure();
230 
231  DenseIntElementsAttr permAttr;
232  if (!matchPattern(op.getPerms(), m_Constant(&permAttr)))
233  return failure();
234  auto permValues = llvm::to_vector<6>(llvm::map_range(
235  // TOSA allows both 32- and 64-bit integer tensors here.
236  permAttr.getValues<APInt>(),
237  [](const APInt &val) { return val.getSExtValue(); }));
238 
239  auto inputType = cast<ShapedType>(op.getInput1().getType());
240 
241  auto resultAttr = transpose(inputValues, inputType, outputType, permValues);
242  rewriter.replaceOpWithNewOp<tosa::ConstOp>(op, outputType, resultAttr);
243  return success();
244  }
245 };

247 struct TosaFoldConstantReciprocal : public OpRewritePattern<ReciprocalOp> {
248 
249  using OpRewritePattern::OpRewritePattern;
250 
251  static APFloat computeReciprocal(const APFloat &floatVal, FloatType floatTy) {
252  auto recipAttr = FloatAttr::get(floatTy, 1.0);
253  APFloat recip = recipAttr.getValue();
254  recip.divide(floatVal, tosaRoundingMode);
255 
256  return recip;
257  }
258 
259  LogicalResult matchAndRewrite(ReciprocalOp recip,
260  PatternRewriter &rewriter) const override {
261  auto inputTensor = recip.getInput1();
262 
263  // Check that we can apply folding
264  auto preCondCheck =
265  notifyIfNotConstantFloatTosaTensor(inputTensor, recip, rewriter);
266  if (failed(preCondCheck)) {
267  return preCondCheck;
268  }
269 
270  // Extract the tensor values
271  DenseElementsAttr inputValues;
272  matchPattern(inputTensor, m_Constant(&inputValues));
273 
274  // Check whether this should be folded.
275  if (!constantUnaryOpShouldBeFolded(recip, inputValues)) {
276  return rewriter.notifyMatchFailure(
277  recip, "Currently, reciprocals will only be folded if the input "
278  "tensor has a single user");
279  }
280 
281  // Create a new tensor with the updated values
282  auto newTensor = applyElementWise<APFloat, APFloat, FloatType>(
283  inputValues, &computeReciprocal,
284  cast<FloatType>(inputValues.getElementType()));
285 
286  // Replace the use of the reciprocal with the transformed tensor
287  rewriter.replaceOpWithNewOp<ConstOp>(recip, newTensor.getType(), newTensor);
288  return success();
289  }
290 };
