54 /// Lower 'arm_sme.zero'. Use 'arm_sme.cast_tile_to_vector' to model the return
55 /// value. The latter is a nop, which should be folded away (e.g. during
56 /// canonicalisation).
57 ///
58 /// BEFORE:
59 /// ```mlir
60 /// %0 = arm_sme.zero : vector<[16]x[16]xi8>
61 /// ```
62 ///
63 /// AFTER:
64 /// ```mlir
65 /// %1 = arm_sme.get_tile_id : i8
66 /// %2 = arm_sme.cast_tile_to_vector %1 : i8 to vector<[16]x[16]xi8>
67 /// "arm_sme.intr.zero"(%c255_i32) : (i32) -> ()
68 /// ```
69 struct ZeroOpConversion : public ConvertOpToLLVMPattern<ZeroOp> {
70  using ConvertOpToLLVMPattern<ZeroOp>::ConvertOpToLLVMPattern;
71 
72  LogicalResult
73  matchAndRewrite(ZeroOp zero, OpAdaptor adaptor,
74  ConversionPatternRewriter &rewriter) const override {
75  auto loc = zero.getLoc();
76 
77  // Get Tile ID for the `zero` intrinsic.
78  // TODO: Map this to a valid `mask` for the `zero` intrinsic.
79  auto tileId = rewriter.create<arm_sme::GetTileID>(
80  loc, zero.getVectorType().getElementType());
81 
82  // Create 'arm_sme.intr.zero' intrinsic to zero ZA.
83  // FIXME: Replace the hard-coded mask with a valid value based
84  // on `tileId`.
85  auto mask = rewriter.create<arith::ConstantOp>(
86  loc, rewriter.getI32Type(), rewriter.getI32IntegerAttr(kZeroZAMask));
87  rewriter.create<arm_sme::aarch64_sme_zero>(loc, mask);
88 
89  // Create `CastTileToVectorOp` to use it as the output
90  rewriter.replaceOpWithNewOp<arm_sme::CastTileToVector>(zero, zero.getType(),
91  tileId);
92 
93  return success();
94  }
95 };

97 /// Extends or truncates `tile`, which should be an `arm_sme::GetTileID` or
98 /// `arm_sme::CastVectorToTile` op returning an 8/16/32/64/128-bit scalar
99 /// integer, to an i32 that can be passed as the `tile` parameter to the SME
100 /// intrinsics. Or returns `tile` if already i32.
101 Value castTileIDToI32(Value tile, Location loc,
102  ConversionPatternRewriter &rewriter) {
103  assert((isa<arm_sme::GetTileID, arm_sme::CastVectorToTile>(
104  tile.getDefiningOp())) &&
105  "expected ArmSME GetTileID or CastVectorToTile op!");
106  unsigned tileElementWidth = tile.getType().getIntOrFloatBitWidth();
107  if (tileElementWidth < 32)
108  return rewriter.create<arith::ExtUIOp>(loc, rewriter.getI32Type(), tile);
109  if (tileElementWidth > 32)
110  return rewriter.create<arith::TruncIOp>(loc, rewriter.getI32Type(), tile);
111  return tile;
112 }

114 /// Lower `arm_sme.load_tile_slice` to SME intrinsics.
115 struct LoadTileSliceToArmSMELowering
116  : public ConvertOpToLLVMPattern<arm_sme::LoadTileSliceOp> {
117  using ConvertOpToLLVMPattern<
118  arm_sme::LoadTileSliceOp>::ConvertOpToLLVMPattern;
119 
120  LogicalResult
121  matchAndRewrite(arm_sme::LoadTileSliceOp loadTileSliceOp,
122  arm_sme::LoadTileSliceOp::Adaptor adaptor,
123  ConversionPatternRewriter &rewriter) const override {
124  auto loc = loadTileSliceOp.getLoc();
125  auto tileType = loadTileSliceOp.getVectorType();
126  auto tileElementType = tileType.getElementType();
127  unsigned tileElementWidth = tileElementType.getIntOrFloatBitWidth();
128 
129  // Create 'arm_sme.cast_vector_to_tile' to get a tile ID for the tile being
130  // loaded to.
131  auto tile = rewriter.create<arm_sme::CastVectorToTile>(
132  loc, rewriter.getIntegerType(tileElementWidth),
133  loadTileSliceOp.getTile());
134 
135  Value ptr = this->getStridedElementPtr(loc, loadTileSliceOp.getMemRefType(),
136  adaptor.getBase(),
137  adaptor.getIndices(), rewriter);
138 
139  auto tileSlice = loadTileSliceOp.getTileSliceIndex();
140 
141  // Cast tile slice to i32 for intrinsic.
142  auto tileSliceI32 = rewriter.create<arith::IndexCastUIOp>(
143  loc, rewriter.getI32Type(), tileSlice);
144 
145  // Create all active predicate mask.
146  auto one = rewriter.create<arith::ConstantOp>(
147  loc, rewriter.getI1Type(),
148  rewriter.getIntegerAttr(rewriter.getI1Type(), 1));
149  auto predTy = VectorType::get(tileType.getShape()[0], rewriter.getI1Type(),
150  /*scalableDims=*/{true});
151  auto allActiveMask = rewriter.create<vector::SplatOp>(loc, predTy, one);
152 
153  auto tileI32 = castTileIDToI32(tile, loc, rewriter);
154  // Create 'arm_sme.intr.ld1*.horiz' intrinsic to load ZA tile slice.
155  switch (tileElementWidth) {
156  default:
157  llvm_unreachable("unexpected element type!");
158  case 8:
159  rewriter.create<arm_sme::aarch64_sme_ld1b_horiz>(loc, allActiveMask, ptr,
160  tileI32, tileSliceI32);
161  break;
162  case 16:
163  rewriter.create<arm_sme::aarch64_sme_ld1h_horiz>(loc, allActiveMask, ptr,
164  tileI32, tileSliceI32);
165  break;
166  case 32:
167  rewriter.create<arm_sme::aarch64_sme_ld1w_horiz>(loc, allActiveMask, ptr,
168  tileI32, tileSliceI32);
169  break;
170  case 64:
171  rewriter.create<arm_sme::aarch64_sme_ld1d_horiz>(loc, allActiveMask, ptr,
172  tileI32, tileSliceI32);
173  break;
174  }
175 
176  // The load intrinsics have no result, replace 'arm_sme.tile_load' with
177  // 'arm_sme.cast_tile_to_vector' to preserve dataflow.
178  rewriter.replaceOpWithNewOp<arm_sme::CastTileToVector>(loadTileSliceOp,
179  tileType, tile);
180 
181  return success();
182  }
183 };

253 void mlir::configureArmSMELegalizeForExportTarget(
254  LLVMConversionTarget &target) {
255  target.addLegalOp<
256  scf::ForOp, scf::YieldOp, arm_sme::CastTileToVector,
257  arm_sme::CastVectorToTile, arm_sme::aarch64_sme_zero,
258  arm_sme::aarch64_sme_str, arm_sme::aarch64_sme_ld1b_horiz,
259  arm_sme::aarch64_sme_ld1h_horiz, arm_sme::aarch64_sme_ld1w_horiz,
260  arm_sme::aarch64_sme_ld1d_horiz, arm_sme::aarch64_sme_st1b_horiz,
261  arm_sme::aarch64_sme_st1h_horiz, arm_sme::aarch64_sme_st1w_horiz,
262  arm_sme::aarch64_sme_st1d_horiz, arm_sme::aarch64_sme_za_enable,
263  arm_sme::aarch64_sme_za_disable>();
264  target.addLegalOp<GetTileID>();
265 
266  // Mark 'func.func' ops as legal if either:
267  // 1. no 'arm_za' function attribute is present.
268  // 2. the 'arm_za' function attribute is present and the first op in the
269  // function is an 'arm_sme::aarch64_sme_za_enable' intrinsic.
270  target.addDynamicallyLegalOp<func::FuncOp>([&](func::FuncOp funcOp) {
271  if (funcOp.isDeclaration())
272  return true;
273  auto firstOp = funcOp.getBody().front().begin();
274  return !funcOp->hasAttr("arm_za") ||
275  isa<arm_sme::aarch64_sme_za_enable>(firstOp);
276  });
277 
278  // Mark 'func.return' ops as legal if either:
279  // 1. no 'arm_za' function attribute is present.
280  // 2. the 'arm_za' function attribute is present and there's a preceding
281  // 'arm_sme::aarch64_sme_za_disable' intrinsic.
282  target.addDynamicallyLegalOp<func::ReturnOp>([&](func::ReturnOp returnOp) {
283  bool hasDisableZA = false;
284  auto funcOp = returnOp->getParentOp();
285  funcOp->walk<WalkOrder::PreOrder>(
286  [&](arm_sme::aarch64_sme_za_disable op) { hasDisableZA = true; });
287  return !funcOp->hasAttr("arm_za") || hasDisableZA;
288  });
289 }
