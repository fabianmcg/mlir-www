41 // Our variation on `AffineParser::{parseBareIdExpr,parseIdentifierDefinition}`
42 OptionalParseResult DimLvlMapParser::parseVar(VarKind vk, bool isOptional,
43  Policy creationPolicy,
44  VarInfo::ID &varID,
45  bool &didCreate) {
46  // Save the current location so that we can have error messages point to
47  // the right place. Note that `Parser::emitWrongTokenError` starts off
48  // with the same location as `AsmParserImpl::getCurrentLocation` returns;
49  // however, `Parser` will then do some various munging with the location
50  // before actually using it, so `AsmParser::emitError` can't quite be used
51  // as a drop-in replacement for `Parser::emitWrongTokenError`.
52  const auto loc = parser.getCurrentLocation();
53 
54  // Several things to note.
55  // (1) the `Parser::isCurrentTokenAKeyword` method checks the exact
56  // same conditions as the `AffineParser.cpp`-static free-function
57  // `isIdentifier` which is used by `AffineParser::parseBareIdExpr`.
58  // (2) the `{Parser,AsmParserImpl}::parseOptionalKeyword(StringRef*)`
59  // methods do the same song and dance about using
60  // `isCurrentTokenAKeyword`, `getTokenSpelling`, et `consumeToken` as we
61  // would want to do if we could use the `Parser` class directly. It
62  // doesn't provide the nice error handling we want, but we can work around
63  // that.
64  StringRef name;
65  if (failed(parser.parseOptionalKeyword(&name))) {
66  // If not actually optional, then `emitError`.
67  ERROR_IF(!isOptional, "expected bare identifier")
68  // If is actually optional, then return the null `OptionalParseResult`.
69  return std::nullopt;
70  }
71 
72  // I don't know if we need to worry about the possibility of the caller
73  // recovering from error and then reusing the `DimLvlMapParser` for subsequent
74  // `parseVar`, but I'm erring on the side of caution by distinguishing
75  // all three possible creation policies.
76  if (const auto res = env.lookupOrCreate(creationPolicy, name, loc, vk)) {
77  varID = res->first;
78  didCreate = res->second;
79  return success();
80  }
81  // TODO(wrengr): these error messages make sense for our intended usage,
82  // but not in general; but it's unclear how best to factor that part out.
83  switch (creationPolicy) {
84  case Policy::MustNot:
85  return parser.emitError(loc, "use of undeclared identifier '" + name + "'");
86  case Policy::May:
87  llvm_unreachable("got nullopt for Policy::May");
88  case Policy::Must:
89  return parser.emitError(loc, "redefinition of identifier '" + name + "'");
90  }
91  llvm_unreachable("unknown Policy");
92 }

137 Var DimLvlMapParser::bindVar(llvm::SMLoc loc, VarInfo::ID id) {
138  MLIRContext *context = parser.getContext();
139  const auto var = env.bindVar(id);
140  const auto &info = std::as_const(env).access(id);
141  const auto name = info.getName();
142  const auto num = *info.getNum();
143  switch (info.getKind()) {
144  case VarKind::Symbol: {
145  const auto affine = getAffineSymbolExpr(num, context);
146  dimsAndSymbols.emplace_back(name, affine);
147  lvlsAndSymbols.emplace_back(name, affine);
148  return var;
149  }
150  case VarKind::Dimension:
151  dimsAndSymbols.emplace_back(name, getAffineDimExpr(num, context));
152  return var;
153  case VarKind::Level:
154  lvlsAndSymbols.emplace_back(name, getAffineDimExpr(num, context));
155  return var;
156  }
157  llvm_unreachable("unknown VarKind");
158 }

185 // FIXME: The forward-declaration of level-vars is a stop-gap workaround
186 // so that we can reuse `AsmParser::parseAffineExpr` in the definition of
187 // `DimLvlMapParser::parseDimSpec`. (In particular, note that all the
188 // variables must be bound before entering `AsmParser::parseAffineExpr`,
189 // since that method requires every variable to already have a fixed/known
190 // `Var::Num`.)
191 //
192 // However, the forward-declaration list duplicates information which is
193 // already encoded by the level-var bindings in `parseLvlSpecList` (namely:
194 // the names of the variables themselves, and the order in which the names
195 // are bound). This redundancy causes bad UX, and also means we must be
196 // sure to verify consistency between the two sources of information.
197 //
198 // Therefore, it would be best to remove the forward-declaration list from
199 // the syntax. This can be achieved by implementing our own version of
200 // `AffineParser::parseAffineExpr` which calls
201 // `parseVarUsage(_,requireKnown=false)` for variables and stores the resulting
202 // `VarInfo::ID` in the expression tree (instead of demanding it be resolved to
203 // some `Var::Num` immediately). This would also enable us to use the `VarEnv`
204 // directly, rather than building the `{dims,lvls}AndSymbols` lists on the
205 // side, and thus would also enable us to avoid the O(n^2) behavior of copying
206 // `DimLvlParser::{dims,lvls}AndSymbols` into `AffineParser::dimsAndSymbols`
207 // every time `AsmParser::parseAffineExpr` is called.
208 ParseResult DimLvlMapParser::parseLvlVarBindingList() {
209  return parser.parseCommaSeparatedList(
210  OpAsmParser::Delimiter::OptionalBraces,
211  [this]() { return ParseResult(parseVarBinding(VarKind::Level)); },
212  " in level declaration list");
213 }

226 ParseResult DimLvlMapParser::parseDimSpec() {
227  // Parse the requisite dim-var binding.
228  const auto varID = parseVarBinding(VarKind::Dimension);
229  FAILURE_IF_FAILED(varID)
230  const DimVar var = env.getVar(*varID).cast<DimVar>();
231 
232  // Parse an optional dimension expression.
233  AffineExpr affine;
234  if (succeeded(parser.parseOptionalEqual())) {
235  // Parse the dim affine expr, with only any lvl-vars in scope.
236  // FIXME(wrengr): This still has the O(n^2) behavior of copying
237  // our `lvlsAndSymbols` into the `AffineParser::dimsAndSymbols`
238  // field every time `parseDimSpec` is called.
239  FAILURE_IF_FAILED(parser.parseAffineExpr(lvlsAndSymbols, affine))
240  }
241  DimExpr expr{affine};
242 
243  // Parse an optional slice.
244  SparseTensorDimSliceAttr slice;
245  if (succeeded(parser.parseOptionalColon())) {
246  const auto loc = parser.getCurrentLocation();
247  Attribute attr;
248  FAILURE_IF_FAILED(parser.parseAttribute(attr))
249  slice = llvm::dyn_cast<SparseTensorDimSliceAttr>(attr);
250  ERROR_IF(!slice, "expected SparseTensorDimSliceAttr")
251  }
252 
253  dimSpecs.emplace_back(var, expr, slice);
254  return success();
255 }

261 ParseResult DimLvlMapParser::parseLvlSpecList() {
262  // This method currently only supports two syntaxes:
263  //
264  // (1) There are no forward-declarations, and no lvl-var bindings:
265  // (d0, d1) -> (d0 : dense, d1 : compressed)
266  // Therefore `parseLvlVarBindingList` didn't bind any lvl-vars, and thus
267  // `parseLvlSpec` will need to use `VarEnv::bindUnusedVar` to ensure that
268  // the level-rank is correct at the end of parsing.
269  //
270  // (2) There are forward-declarations, and every lvl-spec must have
271  // a lvl-var binding:
272  // {l0, l1} (d0 = l0, d1 = l1) -> (l0 = d0 : dense, l1 = d1 : compressed)
273  // However, this introduces duplicate information since the order of
274  // the lvl-vars in `parseLvlVarBindingList` must agree with their order
275  // in the list of lvl-specs. Therefore, `parseLvlSpec` will not call
276  // `VarEnv::bindVar` (since `parseLvlVarBindingList` already did so),
277  // and must also validate the consistency between the two lvl-var orders.
278  const auto declaredLvlRank = env.getRanks().getLvlRank();
279  const bool requireLvlVarBinding = declaredLvlRank != 0;
280  // Have `ERROR_IF` point to the start of the list.
281  const auto loc = parser.getCurrentLocation();
282  const auto res = parser.parseCommaSeparatedList(
283  mlir::OpAsmParser::Delimiter::Paren,
284  [=]() -> ParseResult { return parseLvlSpec(requireLvlVarBinding); },
285  " in level-specifier list");
286  FAILURE_IF_FAILED(res)
287  const auto specLvlRank = lvlSpecs.size();
288  ERROR_IF(requireLvlVarBinding && specLvlRank != declaredLvlRank,
289  "Level-rank mismatch between forward-declarations and specifiers. "
290  "Declared " +
291  Twine(declaredLvlRank) + " level-variables; but got " +
292  Twine(specLvlRank) + " level-specifiers.")
293  return success();
294 }

307 // NOTE: This is factored out as a separate method only because `Var`
308 // lacks a default-ctor, which makes this conditional difficult to inline
309 // at the one call-site.
310 FailureOr<LvlVar>
311 DimLvlMapParser::parseLvlVarBinding(bool requireLvlVarBinding) {
312  // Nothing to parse, just bind an unnamed variable.
313  if (!requireLvlVarBinding)
314  return env.bindUnusedVar(VarKind::Level).cast<LvlVar>();
315 
316  const auto loc = parser.getCurrentLocation();
317  // NOTE: Calling `parseVarUsage` here is semantically inappropriate,
318  // since the thing we're parsing is supposed to be a variable *binding*
319  // rather than a variable *use*. However, the call to `VarEnv::bindVar`
320  // (and its corresponding call to `DimLvlMapParser::recordVarBinding`)
321  // already occured in `parseLvlVarBindingList`, and therefore we must
322  // use `parseVarUsage` here in order to operationally do the right thing.
323  const auto varID = parseVarUsage(VarKind::Level, /*requireKnown=*/true);
324  FAILURE_IF_FAILED(varID)
325  const auto &info = std::as_const(env).access(*varID);
326  const auto var = info.getVar().cast<LvlVar>();
327  const auto forwardNum = var.getNum();
328  const auto specNum = lvlSpecs.size();
329  ERROR_IF(forwardNum != specNum,
330  "Level-variable ordering mismatch. The variable '" + info.getName() +
331  "' was forward-declared as the " + nth(forwardNum) +
332  " level; but is bound by the " + nth(specNum) +
333  " specification.")
334  FAILURE_IF_FAILED(parser.parseEqual())
335  return var;
336 }

338 ParseResult DimLvlMapParser::parseLvlSpec(bool requireLvlVarBinding) {
339  // Parse the optional lvl-var binding. (Actually, `requireLvlVarBinding`
340  // specifies whether that "optional" is actually Must or MustNot.)
341  const auto varRes = parseLvlVarBinding(requireLvlVarBinding);
342  FAILURE_IF_FAILED(varRes)
343  const LvlVar var = *varRes;
344 
345  // Parse the lvl affine expr, with only the dim-vars in scope.
346  AffineExpr affine;
347  // FIXME(wrengr): This still has the O(n^2) behavior of copying
348  // our `dimsAndSymbols` into the `AffineParser::dimsAndSymbols`
349  // field every time `parseLvlSpec` is called.
350  FAILURE_IF_FAILED(parser.parseAffineExpr(dimsAndSymbols, affine))
351  LvlExpr expr{affine};
352 
353  FAILURE_IF_FAILED(parser.parseColon())
354  const auto type = lvlTypeParser.parseLvlType(parser);
355  FAILURE_IF_FAILED(type)
356 
357  lvlSpecs.emplace_back(var, expr, *type);
358  return success();
359 }
