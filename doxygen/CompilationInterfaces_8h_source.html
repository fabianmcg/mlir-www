38 /// This class serves as an opaque interface for passing options to the
39 /// `TargetAttrInterface` methods. Users of this class must implement the
40 /// `classof` method as well as using the macros `MLIR_*_EXPLICIT_TYPE_ID` to
41 /// ensure type safeness. Targets are free to ignore these options.
42 class TargetOptions {
43 public:
44  /// The target representation of the compilation process.
45  typedef enum {
46  offload, /// The process should produce an offloading representation. For
47  /// the NVVM & ROCDL targets this option produces LLVM IR.
48  assembly, /// The process should produce assembly code.
49  binary /// The process should produce a binary.
50  } CompilationTarget;
51 
52  /// Constructor initializing the toolkit path, the list of files to link to,
53  /// extra command line options & the compilation target. The default
54  /// compilation target is `binary`.
55  TargetOptions(StringRef toolkitPath = {},
56  ArrayRef<std::string> linkFiles = {}, StringRef cmdOptions = {},
57  CompilationTarget compilationTarget = binary);
58 
59  /// Returns the typeID.
60  TypeID getTypeID() const;
61 
62  /// Returns the toolkit path.
63  StringRef getToolkitPath() const;
64 
65  /// Returns the files to link to.
66  ArrayRef<std::string> getLinkFiles() const;
67 
68  /// Returns the command line options.
69  StringRef getCmdOptions() const;
70 
71  /// Returns a tokenization of the command line options.
72  std::pair<llvm::BumpPtrAllocator, SmallVector<const char *>>
73  tokenizeCmdOptions() const;
74 
75  /// Returns the compilation target.
76  CompilationTarget getCompilationTarget() const;
77 
78 protected:
79  /// Derived classes must use this constructor to initialize `typeID` to the
80  /// appropiate value: ie. `TargetOptions(TypeID::get<DerivedClass>())`.
81  TargetOptions(TypeID typeID, StringRef toolkitPath = {},
82  ArrayRef<std::string> linkFiles = {}, StringRef cmdOptions = {},
83  CompilationTarget compilationTarget = binary);
84 
85  /// Path to the target toolkit.
86  std::string toolkitPath;
87 
88  /// List of files to link with the LLVM module.
89  SmallVector<std::string> linkFiles;
90 
91  /// An optional set of command line options to be used by the compilation
92  /// process.
93  std::string cmdOptions;
94 
95  /// Compilation process target representation.
96  CompilationTarget compilationTarget;
97 
98 private:
99  TypeID typeID;
100 };
