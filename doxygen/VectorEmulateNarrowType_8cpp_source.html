31 struct ConvertVectorLoad final : OpConversionPattern<vector::LoadOp> {
32  using OpConversionPattern::OpConversionPattern;
33 
34  LogicalResult
35  matchAndRewrite(vector::LoadOp op, OpAdaptor adaptor,
36  ConversionPatternRewriter &rewriter) const override {
37 
38  auto loc = op.getLoc();
39  auto sourceType = cast<MemRefType>(adaptor.getBase().getType());
40  Type oldElementType = op.getType().getElementType();
41  Type newElementType = sourceType.getElementType();
42  int srcBits = oldElementType.getIntOrFloatBitWidth();
43  int dstBits = newElementType.getIntOrFloatBitWidth();
44 
45  if (dstBits % srcBits != 0) {
46  return rewriter.notifyMatchFailure(
47  op, "only dstBits % srcBits == 0 supported");
48  }
49  int scale = dstBits / srcBits;
50 
51  // Adjust the number of elements to load when emulating narrow types,
52  // and then cast back to the original type with vector.bitcast op.
53  // Here only the 1-D vector load is considered, and the N-D memref types
54  // should be linearized.
55  // For example, to emulate i4 to i8, the following op:
56  //
57  // %1 = vector.load %0[%c0, %c0] : memref<3x4xi4>, vector<4xi4>
58  //
59  // can be replaced with
60  //
61  // %1 = vector.load %0[%linear_index] : memref<12xi8>, vector<2xi8>
62  // %2 = vector.bitcast %1 : vector<2xi8> to vector<4xi4>
63  //
64  // TODO: Currently, only the even number of elements loading is supported.
65  // To deal with the odd number of elements, one has to extract the
66  // subvector at the proper offset after bit-casting.
67 
68  auto origElements = op.getVectorType().getNumElements();
69  if (origElements % scale != 0)
70  return failure();
71 
72  auto stridedMetadata = rewriter.create<memref::ExtractStridedMetadataOp>(
73  loc, adaptor.getBase());
74 
75  auto [reinterpret, linearizedOffset] = memref::getLinearizeMemRefAndOffset(
76  loc, sourceType, srcBits, dstBits, adaptor.getIndices(),
77  stridedMetadata, rewriter);
78 
79  auto srcElementType = sourceType.getElementType();
80  auto numElements =
81  static_cast<int>(std::ceil(static_cast<double>(origElements) / scale));
82  auto newLoad = rewriter.create<vector::LoadOp>(
83  loc, VectorType::get(numElements, srcElementType), reinterpret,
84  linearizedOffset);
85 
86  numElements *= scale;
87  auto castType = VectorType::get(numElements, oldElementType);
88  auto bitCast = rewriter.create<vector::BitCastOp>(loc, castType, newLoad);
89 
90  rewriter.replaceOp(op, bitCast->getResult(0));
91  return success();
92  }
93 };
94 } // end anonymous namespace
