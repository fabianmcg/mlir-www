55 class PtxBuilder {
56  NVVM::BasicPtxBuilderInterface op;
57  PatternRewriter &rewriter;
58  std::string asmStr;
59  SmallVector<Value> asmVals;
60  std::string asmConstraints;
61  bool sideEffects;
62  bool hasResult = false;
63 
64  // https://docs.nvidia.com/cuda/inline-ptx-assembly/index.html#constraints
65  char getRegisterType(Type type) {
66  if (type.isInteger(16))
67  return 'h';
68  if (type.isInteger(32))
69  return 'r';
70  if (type.isInteger(64))
71  return 'l';
72  if (type.isF32())
73  return 'f';
74  if (type.isF64())
75  return 'd';
76  if (auto ptr = type.dyn_cast<LLVM::LLVMPointerType>()) {
77  // Shared address spaces is addressed with 32-bit pointers.
78  if (ptr.getAddressSpace() == NVVM::kSharedMemorySpace) {
79  return 'r';
80  }
81  return 'l';
82  }
83  op->emitError() << "Register type could not deduced from MLIR type: "
84  << type;
85  return ' ';
86  }
87 
88  char getRegisterType(Value v) {
89  if (v.getDefiningOp<LLVM::ConstantOp>())
90  return 'n';
91  return getRegisterType(v.getType());
92  }
93 
94 public:
95  PtxBuilder(Operation *op, PatternRewriter &rewriter, std::string ptxAsm,
96  bool sideEffects = false)
97  : op(op), rewriter(rewriter), asmStr(std::move(ptxAsm)),
98  sideEffects(sideEffects) {}
99 
100  void insertValue(Value v, PTXRegisterMod itype = PTXRegisterMod::Read) {
101  LLVM_DEBUG(DBGS() << v << "\t Modifier : " << itype << "\n");
102  auto getModifier = [&]() -> const char * {
103  if (itype == PTXRegisterMod::ReadWrite) {
104  assert(false && "Read-Write modifier is not supported. Try setting the "
105  "same value as Write and Read seperately.");
106  return "+";
107  }
108  if (itype == PTXRegisterMod::Write) {
109  return "=";
110  }
111  return "";
112  };
113  auto addValue = [&](Value v) {
114  if (itype == PTXRegisterMod::Read) {
115  asmVals.push_back(v);
116  return;
117  }
118  if (itype == PTXRegisterMod::ReadWrite)
119  asmVals.push_back(v);
120  hasResult = true;
121  };
122 
123  llvm::raw_string_ostream ss(asmConstraints);
124  // Handle Structs
125  if (auto stype = dyn_cast<LLVM::LLVMStructType>(v.getType())) {
126  if (itype == PTXRegisterMod::Write) {
127  addValue(v);
128  }
129  for (auto [idx, t] : llvm::enumerate(stype.getBody())) {
130  if (itype != PTXRegisterMod::Write) {
131  Value extractValue =
132  rewriter.create<LLVM::ExtractValueOp>(op->getLoc(), v, idx);
133  addValue(extractValue);
134  }
135  if (itype == PTXRegisterMod::ReadWrite) {
136  ss << idx << ",";
137  } else {
138  ss << getModifier() << getRegisterType(t) << ",";
139  }
140  ss.flush();
141  }
142  return;
143  }
144  // Handle Scalars
145  addValue(v);
146  ss << getModifier() << getRegisterType(v) << ",";
147  ss.flush();
148  }
149 
150  LLVM::InlineAsmOp build() {
151  auto asmDialectAttr =
152  LLVM::AsmDialectAttr::get(op->getContext(), LLVM::AsmDialect::AD_ATT);
153 
154  auto resultTypes = op->getResultTypes();
155 
156  // Remove the last comma from the constraints string.
157  if (!asmConstraints.empty() &&
158  asmConstraints[asmConstraints.size() - 1] == ',')
159  asmConstraints.pop_back();
160 
161  // asm keywords expects %, but inline assembly uses $. Replace all % with $
162  std::replace(asmStr.begin(), asmStr.end(), '%', '$');
163 
164  return rewriter.create<LLVM::InlineAsmOp>(
165  op->getLoc(),
166  /*result types=*/resultTypes,
167  /*operands=*/asmVals,
168  /*asm_string=*/llvm::StringRef(asmStr),
169  /*constraints=*/asmConstraints.data(),
170  /*has_side_effects=*/sideEffects,
171  /*is_align_stack=*/false,
172  /*asm_dialect=*/asmDialectAttr,
173  /*operand_attrs=*/ArrayAttr());
174  }
175 
176  void buildAndReplaceOp() {
177  LLVM::InlineAsmOp inlineAsmOp = build();
178  LLVM_DEBUG(DBGS() << "\n Generated PTX \n\t" << inlineAsmOp << "\n");
179  if (inlineAsmOp->getNumResults() == op->getNumResults())
180  rewriter.replaceOp(op, inlineAsmOp);
181  else
182  rewriter.eraseOp(op);
183  }

186 struct PtxLowering
187  : public OpInterfaceRewritePattern<NVVM::BasicPtxBuilderInterface> {
188  using OpInterfaceRewritePattern<
189  NVVM::BasicPtxBuilderInterface>::OpInterfaceRewritePattern;
190 
191  PtxLowering(MLIRContext *context, PatternBenefit benefit = 2)
192  : OpInterfaceRewritePattern(context, benefit) {}
193 
194  LogicalResult matchAndRewrite(NVVM::BasicPtxBuilderInterface op,
195  PatternRewriter &rewriter) const override {
196  if (op.hasIntrinsic()) {
197  LLVM_DEBUG(DBGS() << "Ptx Builder does not lower \n\t" << op << "\n");
198  return failure();
199  }
200 
201  SmallVector<std::pair<Value, PTXRegisterMod>> asmValues;
202  LLVM_DEBUG(DBGS() << op.getPtx() << "\n");
203  PtxBuilder generator(op, rewriter, op.getPtx(), op.hasSideEffect());
204 
205  op.getAsmValues(rewriter, asmValues);
206  for (auto &[asmValue, modifier] : asmValues) {
207  LLVM_DEBUG(DBGSNL() << asmValue << "\t Modifier : " << modifier);
208  generator.insertValue(asmValue, modifier);
209  }
210 
211  generator.buildAndReplaceOp();
212  return success();
213  }
