93 static std::optional<MatchNeg> matchNeg(DimLvlExpr expr) {
94  const auto [lhs, op, rhs] = expr.unpackBinop();
95  if (op == AffineExprKind::Constant) {
96  const auto val = expr.castConstantValue();
97  if (val < 0)
98  return MatchNeg{DimLvlExpr{expr.getExprKind(), AffineExpr()}, val};
99  }
100  if (op == AffineExprKind::Mul)
101  if (const auto rval = rhs.tryGetConstantValue(); rval && *rval < 0)
102  return MatchNeg{lhs, *rval};
103  return std::nullopt;
104 }

106 // A heavily revised version of `AsmPrinter::Impl::printAffineExprInternal`.
107 void DimLvlExpr::printAffineExprInternal(
108  llvm::raw_ostream &os, BindingStrength enclosingTightness) const {
109  const char *binopSpelling = nullptr;
110  switch (getAffineKind()) {
111  case AffineExprKind::SymbolId:
112  os << castSymVar();
113  return;
114  case AffineExprKind::DimId:
115  os << castDimLvlVar();
116  return;
117  case AffineExprKind::Constant:
118  os << castConstantValue();
119  return;
120  case AffineExprKind::Add:
121  binopSpelling = " + "; // N.B., this is unused
122  break;
123  case AffineExprKind::Mul:
124  binopSpelling = " * ";
125  break;
126  case AffineExprKind::FloorDiv:
127  binopSpelling = " floordiv ";
128  break;
129  case AffineExprKind::CeilDiv:
130  binopSpelling = " ceildiv ";
131  break;
132  case AffineExprKind::Mod:
133  binopSpelling = " mod ";
134  break;
135  }
136 
137  if (enclosingTightness == BindingStrength::Strong)
138  os << '(';
139 
140  const auto [lhs, op, rhs] = unpackBinop();
141  if (op == AffineExprKind::Mul && rhs.hasConstantValue(-1)) {
142  // Pretty print `(lhs * -1)` as "-lhs".
143  os << '-';
144  lhs.printStrong(os);
145  } else if (op != AffineExprKind::Add) {
146  // Default rule for tightly binding binary operators.
147  // (Including `Mul` that didn't match the previous rule.)
148  lhs.printStrong(os);
149  os << binopSpelling;
150  rhs.printStrong(os);
151  } else {
152  // Combination of all the special rules for addition/subtraction.
153  lhs.printWeak(os);
154  const auto rx = matchNeg(rhs);
155  os << (rx ? " - " : " + ");
156  const auto &rlhs = rx ? rx->getLHS() : rhs;
157  const auto rrhs = rx ? rx->getRHS() : -1; // value irrelevant when `!rx`
158  const bool nonunit = rrhs != -1; // value irrelevant when `!rx`
159  const bool isStrong =
160  rx && rlhs && (nonunit || rlhs.getAffineKind() == AffineExprKind::Add);
161  if (rlhs)
162  rlhs.printAffineExprInternal(os, BindingStrength{isStrong});
163  if (rx && rlhs && nonunit)
164  os << " * ";
165  if (rx && (!rlhs || nonunit))
166  os << -rrhs;
167  }
168 
169  if (enclosingTightness == BindingStrength::Strong)
170  os << ')';
171 }

250 DimLvlMap::DimLvlMap(unsigned symRank, ArrayRef<DimSpec> dimSpecs,
251  ArrayRef<LvlSpec> lvlSpecs)
252  : symRank(symRank), dimSpecs(dimSpecs), lvlSpecs(lvlSpecs),
253  mustPrintLvlVars(false) {
254  // First, check integrity of the variable-binding structure.
255  // NOTE: This establishes the invariant that calls to `VarSet::add`
256  // below cannot cause OOB errors.
257  assert(isWF());
258 
259  // TODO: Second, we need to infer/validate the `lvlToDim` mapping.
260  // Along the way we should set every `DimSpec::elideExpr` according
261  // to whether the given expression is inferable or not. Notably, this
262  // needs to happen before the code for setting every `LvlSpec::elideVar`,
263  // since if the LvlVar is only used in elided DimExpr, then the
264  // LvlVar should also be elided.
265  // NOTE: Be sure to use `DimLvlMap::setDimExpr` for setting the new exprs,
266  // to ensure that we maintain the invariant established by `isWF` above.
267 
268  // Third, we set every `LvlSpec::elideVar` according to whether that
269  // LvlVar occurs in a non-elided DimExpr (TODO: or CountingExpr).
270  // NOTE: The invariant established by `isWF` ensures that the following
271  // calls to `VarSet::add` cannot raise OOB errors.
272  VarSet usedVars(getRanks());
273  for (const auto &dimSpec : dimSpecs)
274  if (!dimSpec.canElideExpr())
275  usedVars.add(dimSpec.getExpr());
276  for (auto &lvlSpec : this->lvlSpecs) {
277  // Is this LvlVar used in any overt expression?
278  const bool isUsed = usedVars.contains(lvlSpec.getBoundVar());
279  // This LvlVar can be elided iff it isn't overtly used.
280  lvlSpec.setElideVar(!isUsed);
281  // If any LvlVar cannot be elided, then must forward-declare all LvlVars.
282  mustPrintLvlVars = mustPrintLvlVars || isUsed;
283  }
284 }

286 bool DimLvlMap::isWF() const {
287  const auto ranks = getRanks();
288  unsigned dimNum = 0;
289  for (const auto &dimSpec : dimSpecs)
290  if (dimSpec.getBoundVar().getNum() != dimNum++ || !dimSpec.isValid(ranks))
291  return false;
292  assert(dimNum == ranks.getDimRank());
293  unsigned lvlNum = 0;
294  for (const auto &lvlSpec : lvlSpecs)
295  if (lvlSpec.getBoundVar().getNum() != lvlNum++ || !lvlSpec.isValid(ranks))
296  return false;
297  assert(lvlNum == ranks.getLvlRank());
298  return true;
299 }

324 void DimLvlMap::print(llvm::raw_ostream &os, bool wantElision) const {
325  // Symbolic identifiers.
326  // NOTE: Unlike `AffineMap` we place the SymVar bindings before the DimVar
327  // bindings, since the SymVars may occur within DimExprs and thus this
328  // ordering helps reduce potential user confusion about the scope of bidings
329  // (since it means SymVars and DimVars both bind-forward in the usual way,
330  // whereas only LvlVars have different binding rules).
331  if (symRank != 0) {
332  os << "[s0";
333  for (unsigned i = 1; i < symRank; ++i)
334  os << ", s" << i;
335  os << ']';
336  }
337 
338  // LvlVar forward-declarations.
339  if (mustPrintLvlVars) {
340  os << '{';
341  llvm::interleaveComma(
342  lvlSpecs, os, [&](LvlSpec const &spec) { os << spec.getBoundVar(); });
343  os << '}';
344  }
345 
346  // Dimension specifiers.
347  os << '(';
348  llvm::interleaveComma(
349  dimSpecs, os, [&](DimSpec const &spec) { spec.print(os, wantElision); });
350  os << ") -> (";
351  // Level specifiers.
352  wantElision = wantElision && !mustPrintLvlVars;
353  llvm::interleaveComma(
354  lvlSpecs, os, [&](LvlSpec const &spec) { spec.print(os, wantElision); });
355  os << ')';
356 }
