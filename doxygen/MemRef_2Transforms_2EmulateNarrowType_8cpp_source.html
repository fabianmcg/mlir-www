31 /// When data is loaded/stored in `targetBits` granularity, but is used in
32 /// `sourceBits` granularity (`sourceBits` < `targetBits`), the `targetBits` is
33 /// treated as an array of elements of width `sourceBits`.
34 /// Return the bit offset of the value at position `srcIdx`. For example, if
35 /// `sourceBits` equals to 4 and `targetBits` equals to 8, the x-th element is
36 /// located at (x % 2) * 4. Because there are two elements in one i8, and one
37 /// element has 4 bits.
38 static Value getOffsetForBitwidth(Location loc, Value srcIdx, int sourceBits,
39  int targetBits, OpBuilder &builder) {
40  assert(targetBits % sourceBits == 0);
41  IntegerType targetType = builder.getIntegerType(targetBits);
42  IntegerAttr idxAttr =
43  builder.getIntegerAttr(targetType, targetBits / sourceBits);
44  auto idx = builder.create<arith::ConstantOp>(loc, targetType, idxAttr);
45  IntegerAttr srcBitsAttr = builder.getIntegerAttr(targetType, sourceBits);
46  auto srcBitsValue =
47  builder.create<arith::ConstantOp>(loc, targetType, srcBitsAttr);
48  auto m = builder.create<arith::RemUIOp>(loc, srcIdx, idx);
49  return builder.create<arith::MulIOp>(loc, targetType, m, srcBitsValue);
50 }

106 struct ConvertMemRefLoad final : OpConversionPattern<memref::LoadOp> {
107  using OpConversionPattern::OpConversionPattern;
108 
109  LogicalResult
110  matchAndRewrite(memref::LoadOp op, OpAdaptor adaptor,
111  ConversionPatternRewriter &rewriter) const override {
112  Type newTy = getTypeConverter()->convertType(op.getMemRefType());
113  if (!newTy) {
114  return rewriter.notifyMatchFailure(
115  op->getLoc(), llvm::formatv("failed to convert memref type: {0}",
116  op.getMemRefType()));
117  }
118 
119  if (op.getMemRefType() == newTy)
120  return failure();
121 
122  auto loc = op.getLoc();
123  auto sourceType = cast<MemRefType>(adaptor.getMemref().getType());
124  unsigned sourceRank = sourceType.getRank();
125  SmallVector<Value> indices = adaptor.getIndices();
126  assert(indices.size() == sourceRank);
127 
128  auto srcElementType = sourceType.getElementType();
129  auto oldElementType = op.getMemRefType().getElementType();
130  int srcBits = oldElementType.getIntOrFloatBitWidth();
131  int dstBits = srcElementType.getIntOrFloatBitWidth();
132  if (dstBits % srcBits != 0) {
133  return rewriter.notifyMatchFailure(
134  op, "only dstBits % srcBits == 0 supported");
135  }
136 
137  auto stridedMetadata = rewriter.create<memref::ExtractStridedMetadataOp>(
138  loc, adaptor.getMemref());
139 
140  Value newLoad, lastIdx;
141  if (sourceRank == 0) {
142  newLoad = rewriter.create<memref::LoadOp>(
143  loc, srcElementType, adaptor.getMemref(), adaptor.getIndices());
144 
145  lastIdx = stridedMetadata.getOffset();
146  } else {
147  auto [reinterpret, linearizedOffset] =
148  memref::getLinearizeMemRefAndOffset(loc, sourceType, srcBits, dstBits,
149  adaptor.getIndices(),
150  stridedMetadata, rewriter);
151 
152  newLoad = rewriter.create<memref::LoadOp>(loc, srcElementType,
153  reinterpret, linearizedOffset);
154 
155  lastIdx = adaptor.getIndices().back();
156  }
157 
158  // Get the offset and shift the bits to the rightmost.
159  // Note, currently only the big-endian is supported.
160  auto castLastIdx =
161  rewriter.create<arith::IndexCastUIOp>(loc, srcElementType, lastIdx);
162 
163  Value BitwidthOffset =
164  getOffsetForBitwidth(loc, castLastIdx, srcBits, dstBits, rewriter);
165  auto bitsLoad =
166  rewriter.create<arith::ShRSIOp>(loc, newLoad, BitwidthOffset);
167 
168  // Get the corresponding bits. If the arith computation bitwidth equals
169  // to the emulated bitwidth, we apply a mask to extract the low bits.
170  // It is not clear if this case actually happens in practice, but we keep
171  // the operations just in case. Otherwise, if the arith computation bitwidth
172  // is different from the emulated bitwidth we truncate the result.
173  Operation *result;
174  auto resultTy = getTypeConverter()->convertType(oldElementType);
175  if (resultTy == srcElementType) {
176  auto mask = rewriter.create<arith::ConstantOp>(
177  loc, srcElementType,
178  rewriter.getIntegerAttr(srcElementType, (1 << srcBits) - 1));
179 
180  result = rewriter.create<arith::AndIOp>(loc, bitsLoad, mask);
181  } else {
182  result = rewriter.create<arith::TruncIOp>(loc, resultTy, bitsLoad);
183  }
184 
185  rewriter.replaceOp(op, result->getResult(0));
186  return success();
187  }
188 };
189 } // end anonymous namespace

205 void memref::populateMemRefNarrowTypeEmulationConversions(
206  arith::NarrowTypeEmulationConverter &typeConverter) {
207  typeConverter.addConversion(
208  [&typeConverter](MemRefType ty) -> std::optional<Type> {
209  auto intTy = dyn_cast<IntegerType>(ty.getElementType());
210  if (!intTy)
211  return ty;
212 
213  unsigned width = intTy.getWidth();
214  unsigned loadStoreWidth = typeConverter.getLoadStoreBitwidth();
215  if (width >= loadStoreWidth)
216  return ty;
217 
218  auto newElemTy = IntegerType::get(ty.getContext(), loadStoreWidth,
219  intTy.getSignedness());
220  if (!newElemTy)
221  return std::nullopt;
222 
223  return ty.cloneWith(std::nullopt, newElemTy);
224  });
225 }
