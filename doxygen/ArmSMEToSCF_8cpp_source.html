29 /// Adjusts `indices` as follows for a given tile slice and returns them in
30 /// `outIndices`:
31 /// rank 1: (indices[0] + (tileSliceIndex * tileSliceNumElts))
32 /// rank 2: (indices[0] + tileSliceIndex, indices[1])
33 void getMemrefIndices(ValueRange indices, unsigned rank, Value tileSliceIndex,
34  Value tileSliceNumElts,
35  SmallVectorImpl<Value> &outIndices, Location loc,
36  PatternRewriter &rewriter) {
37  assert((rank == 1 || rank == 2) && "memref has unexpected rank!");
38 
39  auto tileSliceOffset = tileSliceIndex;
40  if (rank == 1)
41  tileSliceOffset =
42  rewriter.create<arith::MulIOp>(loc, tileSliceOffset, tileSliceNumElts);
43 
44  auto baseIndexPlusTileSliceOffset =
45  rewriter.create<arith::AddIOp>(loc, indices[0], tileSliceOffset);
46  outIndices.push_back(baseIndexPlusTileSliceOffset);
47 
48  if (rank == 2)
49  outIndices.push_back(indices[1]);
50 }

52 /// Lower `arm_sme.tile_load` to a loop over the tile slices and load each slice
53 /// using `arm_sme.load_tile_slice`.
54 ///
55 /// BEFORE:
56 /// ```mlir
57 /// %tile = arm_sme.tile_load %src[%c0, %c0] :
58 /// memref<?x?xi32>, vector<[4]x[4]xi32>
59 /// ```
60 ///
61 /// AFTER:
62 /// ```mlir
63 /// %tile_id = arm_sme.get_tile_id : i32
64 /// %tile = arm_sme.cast_tile_to_vector %tile_id : i32 to vector<[4]x[4]xi32>
65 /// %vscale = vector.vscale
66 /// %c0 = arith.constant 0 : index
67 /// %c1 = arith.constant 1 : index
68 /// %min_svl_s = arith.constant 4 : index
69 /// %svl_s = arith.muli %min_svl_s, %vscale : index
70 /// scf.for %tile_slice_idx = %c0 to %svl_s step %c1 {
71 /// %tile_update = arm_sme.load_tile_slice %src[%tile_slice_idx],
72 /// %tile, %tile_slice_idx : memref<?x?xi32>, vector<[4]x[4]xi32>
73 /// }
74 /// ```
75 struct TileLoadOpConversion : public OpRewritePattern<arm_sme::TileLoadOp> {
76  using OpRewritePattern<arm_sme::TileLoadOp>::OpRewritePattern;
77 
78  LogicalResult matchAndRewrite(arm_sme::TileLoadOp tileLoadOp,
79  PatternRewriter &rewriter) const override {
80  OpBuilder::InsertionGuard g(rewriter);
81  auto loc = tileLoadOp.getLoc();
82  auto tileType = tileLoadOp.getVectorType();
83  auto tileElementType = tileType.getElementType();
84  unsigned tileElementWidth = tileElementType.getIntOrFloatBitWidth();
85 
86  // Create 'arm_sme.get_tile' op.
87  auto tileId = rewriter.create<arm_sme::GetTileID>(
88  loc, rewriter.getIntegerType(tileElementWidth));
89 
90  // Create `arm_sme.cast_tile_to_vector` to cast tile ID to a vector type to
91  // use as input tile to 'arm_sme.load_tile_slice' ops.
92  auto tile =
93  rewriter.create<arm_sme::CastTileToVector>(loc, tileType, tileId);
94 
95  // Create a loop that loads each ZA tile slice from memory.
96  auto step = rewriter.create<arith::ConstantIndexOp>(loc, 1);
97  auto minTileSlices = rewriter.create<arith::ConstantIndexOp>(
98  loc, arm_sme::getSMETileSliceMinNumElts(tileElementType));
99  auto vscale =
100  rewriter.create<vector::VectorScaleOp>(loc, rewriter.getIndexType());
101  auto lowerBound = rewriter.create<arith::ConstantIndexOp>(loc, 0);
102  // This describes both the number of ZA tile slices and the number of
103  // elements in a vector of SVL bits for a given element type (SVL_B, SVL_H,
104  // ..., SVL_Q).
105  auto numTileSlices =
106  rewriter.create<arith::MulIOp>(loc, minTileSlices, vscale);
107  auto forOp =
108  rewriter.create<scf::ForOp>(loc, lowerBound, numTileSlices, step);
109 
110  rewriter.setInsertionPointToStart(forOp.getBody());
111 
112  // Create 'arm_sme.load_tile_slice' to load tile slice from memory into
113  // tile.
114  SmallVector<Value> memrefIndices;
115  auto tileSliceIndex = forOp.getInductionVar();
116  getMemrefIndices(tileLoadOp.getIndices(),
117  tileLoadOp.getMemRefType().getRank(), tileSliceIndex,
118  numTileSlices, memrefIndices, loc, rewriter);
119  rewriter.create<arm_sme::LoadTileSliceOp>(loc, tileType,
120  tileLoadOp.getBase(), tile,
121  memrefIndices, tileSliceIndex);
122 
123  rewriter.setInsertionPointAfter(forOp);
124 
125  // Replace 'arm_sme.tile_load' with the tile.
126  rewriter.replaceOp(tileLoadOp, tile);
127 
128  return success();
129  }
130 };

132 /// Lower `arm_sme.tile_store` to a loop over the tile slices and store each
133 /// slice using `arm_sme.store_tile_slice`.
134 ///
135 /// BEFORE:
136 /// ```mlir
137 /// arm_sme.tile_store %tile, %dest[%c0, %c0]
138 /// : memref<?x?xi32>, vector<[4]x[4]xi32
139 /// ```
140 ///
141 /// AFTER:
142 /// ```mlir
143 /// %vscale = vector.vscale
144 /// %c0 = arith.constant 0 : index
145 /// %c1 = arith.constant 1 : index
146 /// %min_svl_s = arith.constant 4 : index
147 /// %svl_s = arith.muli %min_svl_s, %vscale : index
148 /// scf.for %tile_slice_idx = %c0 to %svl_s step %c1 {
149 /// arm_sme.store_tile_slice %tile, %tile_slice_idx, %dest[%tile_slice_idx]
150 /// : memref<?x?xi32>, vector<[4]x[4]xi32>
151 /// }
152 /// ```
153 struct TileStoreOpConversion : public OpRewritePattern<arm_sme::TileStoreOp> {
154  using OpRewritePattern<arm_sme::TileStoreOp>::OpRewritePattern;
155 
156  LogicalResult matchAndRewrite(arm_sme::TileStoreOp tileStoreOp,
157  PatternRewriter &rewriter) const override {
158  OpBuilder::InsertionGuard g(rewriter);
159  auto loc = tileStoreOp.getLoc();
160  auto tileType = tileStoreOp.getVectorType();
161  auto tileElementType = tileType.getElementType();
162 
163  // Create a loop that stores each ZA tile slice from memory.
164  auto step = rewriter.create<arith::ConstantIndexOp>(loc, 1);
165  auto minTileSlices = rewriter.create<arith::ConstantIndexOp>(
166  loc, arm_sme::getSMETileSliceMinNumElts(tileElementType));
167  auto vscale =
168  rewriter.create<vector::VectorScaleOp>(loc, rewriter.getIndexType());
169  auto lowerBound = rewriter.create<arith::ConstantIndexOp>(loc, 0);
170  // This describes both the number of ZA tile slices and the number of
171  // elements in a vector of SVL bits for a given element type (SVL_B, SVL_H,
172  // ..., SVL_Q).
173  auto numTileSlices =
174  rewriter.create<arith::MulIOp>(loc, minTileSlices, vscale);
175  auto forOp =
176  rewriter.create<scf::ForOp>(loc, lowerBound, numTileSlices, step);
177 
178  rewriter.setInsertionPointToStart(forOp.getBody());
179 
180  SmallVector<Value> memrefIndices;
181  auto tileSliceIndex = forOp.getInductionVar();
182  getMemrefIndices(tileStoreOp.getIndices(),
183  tileStoreOp.getMemRefType().getRank(), tileSliceIndex,
184  numTileSlices, memrefIndices, loc, rewriter);
185  rewriter.replaceOpWithNewOp<arm_sme::StoreTileSliceOp>(
186  tileStoreOp, tileStoreOp.getValueToStore(), tileSliceIndex,
187  tileStoreOp.getBase(), memrefIndices);
188 
189  return success();
190  }

202 struct ConvertArmSMEToSCFPass
203  : public impl::ConvertArmSMEToSCFBase<ConvertArmSMEToSCFPass> {
204  void runOnOperation() override {
205  RewritePatternSet patterns(&getContext());
206  ConversionTarget target(getContext());
207  populateArmSMEToSCFConversionPatterns(patterns);
208  target.addLegalDialect<arm_sme::ArmSMEDialect, vector::VectorDialect,
209  arith::ArithDialect, scf::SCFDialect>();
210  target.addIllegalOp<arm_sme::TileLoadOp, arm_sme::TileStoreOp>();
211  if (failed(applyPartialConversion(getOperation(), target,
212  std::move(patterns))))
213  signalPassFailure();
214  }
215 };
