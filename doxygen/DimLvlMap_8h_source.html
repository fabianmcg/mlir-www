196 /// The full `dimVar = dimExpr : dimSlice` specification for a given dimension.
197 class DimSpec final {
198  /// The dimension-variable bound by this specification.
199  DimVar var;
200  /// The dimension-expression. The `DimSpec` ctor treats this field
201  /// as optional; whereas the `DimLvlMap` ctor will fill in (or verify)
202  /// the expression via function-inversion inference.
203  DimExpr expr;
204  /// Can the `expr` be elided when printing? The `DimSpec` ctor assumes
205  /// not (though if `expr` is null it will elide printing that); whereas
206  /// the `DimLvlMap` ctor will reset it as appropriate.
207  bool elideExpr = false;
208  /// The dimension-slice; optional, default is null.
209  SparseTensorDimSliceAttr slice;
210 
211 public:
212  DimSpec(DimVar var, DimExpr expr, SparseTensorDimSliceAttr slice);
213 
214  MLIRContext *tryGetContext() const { return expr.tryGetContext(); }
215 
216  constexpr DimVar getBoundVar() const { return var; }
217  bool hasExpr() const { return static_cast<bool>(expr); }
218  constexpr DimExpr getExpr() const { return expr; }
219  void setExpr(DimExpr newExpr) {
220  assert(!hasExpr());
221  expr = newExpr;
222  }
223  constexpr bool canElideExpr() const { return elideExpr; }
224  void setElideExpr(bool b) { elideExpr = b; }
225  constexpr SparseTensorDimSliceAttr getSlice() const { return slice; }
226 
227  /// Checks whether the variables bound/used by this spec are valid with
228  /// respect to the given ranks. Note that null `DimExpr` is considered
229  /// to be vacuously valid, and therefore calling `setExpr` invalidates
230  /// the result of this predicate.
231  [[nodiscard]] bool isValid(Ranks const &ranks) const;
232 
233  // TODO(wrengr): Use it or loose it.
234  bool isFunctionOf(Var var) const;
235  bool isFunctionOf(VarSet const &vars) const;
236  void getFreeVars(VarSet &vars) const;
237 
238  void print(llvm::raw_ostream &os, bool wantElision = true) const;
239  void print(AsmPrinter &printer, bool wantElision = true) const;
240  void dump() const;
241 };

247 /// The full `lvlVar = lvlExpr : lvlType` specification for a given level.
248 class LvlSpec final {
249  /// The level-variable bound by this specification.
250  LvlVar var;
251  /// Can the `var` be elided when printing? The `LvlSpec` ctor assumes not;
252  /// whereas the `DimLvlMap` ctor will reset this as appropriate.
253  bool elideVar = false;
254  /// The level-expression.
255  //
256  // NOTE: For now we use `LvlExpr` because all level-expressions must be
257  // `AffineExpr`; however, in the future we will also want to allow "counting
258  // expressions", and potentially other kinds of non-affine level-expressions.
259  // Which kinds of `DimLvlExpr` are allowed will depend on the `DimLevelType`,
260  // so we may consider defining another class for pairing those two together
261  // to ensure that the pair is well-formed.
262  LvlExpr expr;
263  /// The level-type (== level-format + lvl-properties).
264  DimLevelType type;
265 
266 public:
267  LvlSpec(LvlVar var, LvlExpr expr, DimLevelType type);
268 
269  MLIRContext *getContext() const {
270  MLIRContext *ctx = expr.tryGetContext();

272  return ctx;
273  }

276  constexpr bool canElideVar() const { return elideVar; }
277  void setElideVar(bool b) { elideVar = b; }
278  constexpr LvlExpr getExpr() const { return expr; }
