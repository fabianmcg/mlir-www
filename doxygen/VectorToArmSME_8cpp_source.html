18 /// Returns true if 'val' is a splat of zero, false otherwise.
19 static bool isSplatZero(Type elemType, DenseElementsAttr val) {
20  if (llvm::isa<FloatType>(elemType))
21  return val && val.isSplat() && val.getSplatValue<APFloat>().isZero();
22  if (llvm::isa<IntegerType>(elemType))
23  return val && val.isSplat() && val.getSplatValue<APInt>().isZero();
24  return false;
25 }

29 /// Conversion pattern for vector.transfer_write.
30 ///
31 /// vector.transfer_write %vector, %source[%c0, %c0] : vector<[16]x[16]xi8>,
32 /// memref<?x?xi8>
33 ///
34 /// is converted to:
35 ///
36 /// arm_sme.tile_store %vector, %source[%c0, %c0] : memref<?x?xi8>,
37 /// vector<[16]x[16]xi8>
38 struct TransferWriteToArmSMELowering
39  : public OpRewritePattern<vector::TransferWriteOp> {
40  using OpRewritePattern<vector::TransferWriteOp>::OpRewritePattern;
41 
42  LogicalResult matchAndRewrite(vector::TransferWriteOp writeOp,
43  PatternRewriter &rewriter) const final {
44  auto vType = writeOp.getVectorType();
45  if (!arm_sme::isValidSMETileVectorType(vType))
46  return failure();
47 
48  if (!llvm::isa<MemRefType>(writeOp.getSource().getType()))
49  return failure();
50 
51  rewriter.replaceOpWithNewOp<arm_sme::TileStoreOp>(
52  writeOp, writeOp.getVector(), writeOp.getSource(),
53  writeOp.getIndices());
54  return success();
55  }
56 };

90 /// Conversion pattern for dense arith.constant.
91 struct ConstantOpToArmSMELowering : public OpRewritePattern<arith::ConstantOp> {
92  using OpRewritePattern<arith::ConstantOp>::OpRewritePattern;
93 
94  LogicalResult matchAndRewrite(arith::ConstantOp constantOp,
95  PatternRewriter &rewriter) const final {
96  auto vType = dyn_cast<VectorType>(constantOp.getType());
97  if (!vType || !arm_sme::isValidSMETileVectorType(vType))
98  return failure();
99 
100  auto denseAttr = dyn_cast<DenseElementsAttr>(constantOp.getValueAttr());
101  if (!denseAttr || !isSplatZero(vType.getElementType(), denseAttr))
102  return failure();
103 
104  rewriter.replaceOpWithNewOp<arm_sme::ZeroOp>(constantOp, vType);
105 
106  return success();
107  }
108 };
