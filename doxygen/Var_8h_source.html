23 // Throughout this namespace we use the name `isWF` (is "well-formed")
24 // for predicates that detect intrinsic structural integrity criteria,
25 // and hence which should always be assertively true. Whereas we reserve
26 // the name `isValid` for predicates that detect extrinsic semantic
27 // integrity criteria, and hence which may legitimately return false even
28 // in well-formed programs. Moreover, "validity" is often a relational
29 // or contextual property, and therefore the same term may be considered
30 // valid in one context yet invalid in another.
31 //
32 // As an example of why we make this distinction, consider `Var`.
33 // A variable is well-formed if its kind and identifier are both well-formed;
34 // this can be checked locally, and the resulting truth-value holds globally.
35 // Whereas, a variable is valid with respect to a particular `Ranks` only if
36 // it is within bounds; and a variable is valid with respect to a particular
37 // `DimLvlMap` only if the variable is bound and all uses of the variable
38 // are within the scope of that binding.
39 
40 // Throughout this namespace we use `enum class` types to form "newtypes".
41 // The enum-based implementation of newtypes only serves to block implicit
42 // conversions; it cannot enforce any wellformedness constraints, since
43 // `enum class` permits using direct-list-initialization to construct
44 // arbitrary values[1]. Consequently, we use the syntax "`E{u}`" whenever
45 // we intend that ctor to be a noop (i.e., `std::is_same_v<decltype(u),
46 // std::underlying_type_t<E>>`), since the compiler will ensure that that's
47 // the case. Whereas we only use the "`static_cast<E>(u)`" syntax when we
48 // specifically intend to introduce conversions.
49 //
50 // [1]:
51 // <https://en.cppreference.com/w/cpp/language/enum#enum_relaxed_init_cpp17>

54 /// The three kinds of variables that `Var` can be.
55 ///
56 /// NOTE: The numerical values used to represent this enum should be
57 /// treated as an implementation detail, not as part of the API. In the
58 /// API below we use the canonical ordering `{Symbol,Dimension,Level}` even
59 /// though that does not agree with the numerical ordering of the numerical
60 /// representation.
61 enum class VarKind { Symbol = 1, Dimension = 0, Level = 2 };
62 
63 [[nodiscard]] constexpr bool isWF(VarKind vk) {
64  const auto vk_ = to_underlying(vk);
65  return 0 <= vk_ && vk_ <= 2;
66 }
67 
68 /// Swaps `Dimension` and `Level`, but leaves `Symbol` the same.
69 constexpr VarKind flipVarKind(VarKind vk) {
70  return VarKind{2 - to_underlying(vk)};
71 }
72 static_assert(flipVarKind(VarKind::Symbol) == VarKind::Symbol &&
73  flipVarKind(VarKind::Dimension) == VarKind::Level &&
74  flipVarKind(VarKind::Level) == VarKind::Dimension);

76 /// Gets the ASCII character used as the prefix when printing `Var`.
77 constexpr char toChar(VarKind vk) {
78  // If `isWF(vk)` then this computation's intermediate results are always
79  // in the range [-44..126] (where that lower bound is under worst-case
80  // rearranging of the expression); and `int_fast8_t` is the fastest type
81  // which can support that range without over-/underflow.
82  const auto vk_ = static_cast<int_fast8_t>(to_underlying(vk));
83  return static_cast<char>(100 + vk_ * (26 - vk_ * 11));
84 }
85 static_assert(toChar(VarKind::Symbol) == 's' &&
86  toChar(VarKind::Dimension) == 'd' &&
87  toChar(VarKind::Level) == 'l');

95 /// A concrete variable, to be used in our variant of `AffineExpr`.
96 class Var {
97  // Design Note: This class makes several distinctions which may at first
98  // seem unnecessary but are in fact needed for implementation reasons.
99  // These distinctions are summarized as follows:
100  //
101  // * `Var`
102  // Client-facing class for `VarKind` + `Var::Num` pairs, with RTTI
103  // support for subclasses with a fixed `VarKind`.
104  // * `Var::Num`
105  // Client-facing typedef for the type of variable numbers; defined
106  // so that client code can use it to disambiguate/document when things
107  // are intended to be variable numbers, as opposed to some other thing
108  // which happens to be represented as `unsigned`.
109  // * `Var::Storage`
110  // Private typedef for the storage of `Var::Impl`; defined only because
111  // it's also needed for defining `kMaxNum`. Note that this type must be
112  // kept distinct from `Var::Num`: not only can they be different C++ types
113  // (even though they currently happen to be the same), but also because
114  // they use different bitwise representations.
115  // * `Var::Impl`
116  // The underlying implementation of `Var`; needed by RTTI to serve as
117  // an intermediary between `Var` and `Var::Storage`. That is, we want
118  // the RTTI methods to select the `U(Var::Impl)` ctor, without any
119  // possibility of confusing that with the `U(Var::Num)` ctor nor with
120  // the copy-ctor. (Although the `U(Var::Impl)` ctor is effectively
121  // identical to the copy-ctor, it doesn't have the type that C++ expects
122  // for a copy-ctor.)
123  //
124  // TODO: See if it'd be cleaner to use "llvm/ADT/Bitfields.h" in lieu
125  // of doing our own bitbashing (though that seems to only be used by LLVM
126  // for defining machine/assembly ops, and not anywhere else in LLVM/MLIR).
127 public:
128  /// Typedef for the type of variable numbers.
129  using Num = unsigned;
130 
131 private:
132  /// Typedef for the underlying storage of `Var::Impl`.
133  using Storage = unsigned;
134 
135  /// The largest `Var::Num` supported by `Var`/`Var::Impl`/`Var::Storage`.
136  /// Two low-order bits are reserved for storing the `VarKind`,
137  /// and one high-order bit is reserved for future use (e.g., to support
138  /// `DenseMapInfo<Var>` while maintaining the usual numeric values for
139  /// "empty" and "tombstone").
140  static constexpr Num kMaxNum =
141  static_cast<Num>(std::numeric_limits<Storage>::max() >> 3);
142 
143 public:
144  /// Checks whether the number would be accepted by `Var(VarKind,Var::Num)`.
145  //
146  // This must be public for `VarInfo` to use it (whereas we don't want
147  // to expose the `impl` field via friendship).
148  [[nodiscard]] static constexpr bool isWF_Num(Num n) { return n <= kMaxNum; }
149 
150 protected:
151  /// The underlying implementation of `Var`. Note that this must be kept
152  /// distinct from `Var` itself, since we want to ensure that the RTTI
153  /// methods will select the `U(Var::Impl)` ctor rather than selecting
154  /// the `U(Var::Num)` ctor.
155  class Impl final {
156  Storage data;
157 
158  public:
159  constexpr Impl(VarKind vk, Num n)
160  : data((static_cast<Storage>(n) << 2) |
161  static_cast<Storage>(to_underlying(vk))) {
162  assert(isWF(vk) && "unknown VarKind");
163  assert(isWF_Num(n) && "Var::Num is too large");
164  }
165  constexpr bool operator==(Impl other) const { return data == other.data; }
166  constexpr bool operator!=(Impl other) const { return !(*this == other); }
167  constexpr VarKind getKind() const { return static_cast<VarKind>(data & 3); }
168  constexpr Num getNum() const { return static_cast<Num>(data >> 2); }
169  };
170  static_assert(IsZeroCostAbstraction<Impl>);
171 
172 private:
173  Impl impl;
174 
175 protected:
176  /// Protected ctor for the RTTI methods to use.
177  constexpr explicit Var(Impl impl) : impl(impl) {}
178 
179 public:
180  constexpr Var(VarKind vk, Num n) : impl(Impl(vk, n)) {}
181  Var(AffineSymbolExpr sym) : Var(VarKind::Symbol, sym.getPosition()) {}
182  // TODO(wrengr): Should make the first argument an `ExprKind` instead...?
183  Var(VarKind vk, AffineDimExpr var) : Var(vk, var.getPosition()) {
184  assert(vk != VarKind::Symbol);
185  }
186 
187  constexpr bool operator==(Var other) const { return impl == other.impl; }
188  constexpr bool operator!=(Var other) const { return !(*this == other); }
189 
190  constexpr VarKind getKind() const { return impl.getKind(); }
191  constexpr Num getNum() const { return impl.getNum(); }
192 
193  template <typename U>
194  constexpr bool isa() const;
195  template <typename U>
196  constexpr U cast() const;
197  template <typename U>
198  constexpr std::optional<U> dyn_cast() const;
199 
200  void print(llvm::raw_ostream &os) const;
201  void print(AsmPrinter &printer) const;
202  void dump() const;
203 };

242 template <typename U>
243 constexpr bool Var::isa() const {
244  if constexpr (std::is_same_v<U, SymVar>)
245  return getKind() == VarKind::Symbol;
246  if constexpr (std::is_same_v<U, DimVar>)
247  return getKind() == VarKind::Dimension;
248  if constexpr (std::is_same_v<U, LvlVar>)
249  return getKind() == VarKind::Level;
250  // NOTE: The `AffineExpr::isa` implementation doesn't have a fallthrough
251  // case returning `false`; wrengr guesses that's so things will fail
252  // to compile whenever `!std::is_base_of<Var, U>`. Though it's unclear
253  // why they implemented it that way rather than using SFINAE for that,
254  // especially since it would give better error messages.
255 }

311 /// Efficient representation of a set of `Var`.
312 ///
313 /// NOTE: For the `contains`/`occursIn` methods: if variables occurring in
314 /// the method parameter are OOB for the `VarSet`, then these methods will
315 /// always return false. However, for the `add` methods: OOB parameters

348 /// A record of metadata for/about a variable, used by `VarEnv`.
349 /// The principal goal of this record is to enable `VarEnv` to be used for
350 /// incremental parsing; in particular, `VarInfo` allows the `Var::Num` to
351 /// remain unknown, since each record is instead identified by `VarInfo::ID`.
352 /// Therefore the `VarEnv` can freely allocate `VarInfo::ID` in whatever
353 /// order it likes, irrespective of the binding order (`Var::Num`) of the
354 /// associated variable.
355 class VarInfo final {
356 public:
357  /// Newtype for unique identifiers of `VarInfo` records, to ensure
358  /// they aren't confused with `Var::Num`.
359  enum class ID : unsigned {};
360 
361 private:
362  StringRef name; // The bare-id used in the MLIR source.
363  llvm::SMLoc loc; // The location of the first occurence.
364  ID id; // The unique `VarInfo`-identifier.
365  std::optional<Var::Num> num; // The unique `Var`-identifier (if resolved).
366  VarKind kind; // The kind of variable.
367 
368 public:
369  constexpr VarInfo(ID id, StringRef name, llvm::SMLoc loc, VarKind vk,
370  std::optional<Var::Num> n = {})
371  : name(name), loc(loc), id(id), num(n), kind(vk) {
372  assert(!name.empty() && "null StringRef");
373  assert(loc.isValid() && "null SMLoc");
374  assert(isWF(vk) && "unknown VarKind");
375  assert((!n || Var::isWF_Num(*n)) && "Var::Num is too large");
376  }
377 
378  constexpr StringRef getName() const { return name; }
379  constexpr llvm::SMLoc getLoc() const { return loc; }
380  Location getLocation(AsmParser &parser) const {
381  return parser.getEncodedSourceLoc(loc);
382  }
383  constexpr ID getID() const { return id; }
384  constexpr VarKind getKind() const { return kind; }
385  constexpr std::optional<Var::Num> getNum() const { return num; }
386  constexpr bool hasNum() const { return num.has_value(); }
387  void setNum(Var::Num n);
388  constexpr Var getVar() const {
389  assert(hasNum());
390  return Var(kind, *num);
391  }
392  constexpr std::optional<Var> tryGetVar() const {
393  return num ? std::make_optional(Var(kind, *num)) : std::nullopt;
394  }
395 };

405 class VarEnv final {
406  /// Map from `VarKind` to the next free `Var::Num`; used by `bindVar`.
407  VarKindArray<Var::Num> nextNum;
408  /// Map from `VarInfo::ID` to shared storage for the actual `VarInfo` objects.

410  /// Map from variable names to their `VarInfo::ID`.
411  llvm::StringMap<VarInfo::ID> ids;

413  VarInfo::ID nextID() const { return static_cast<VarInfo::ID>(vars.size()); }
414 
415 public:
416  VarEnv() : nextNum(0) {}
417 
418  /// Gets the underlying storage for the `VarInfo` identified by
419  /// the `VarInfo::ID`.
420  ///
421  /// NOTE: The returned reference can become dangling if the `VarEnv`
422  /// object is mutated during the lifetime of the pointer. Therefore,
423  /// client code should not store the reference nor otherwise allow it
424  /// to live too long.
425  //
426  // FUTURE_CL(wrengr): Consider trying to define/use a nested class
427  // `struct{VarEnv*; VarInfo::ID}` akin to `BitVector::reference`.
428  VarInfo const &access(VarInfo::ID id) const {
429  // `SmallVector::operator[]` already asserts the index is in-bounds.
430  return vars[to_underlying(id)];
431  }

433  return oid ? &access(*oid) : nullptr;
434  }

438  return const_cast<VarInfo &>(std::as_const(*this).access(id));
439  }
440  VarInfo *access(std::optional<VarInfo::ID> oid) {
441  return const_cast<VarInfo *>(std::as_const(*this).access(oid));
442  }
443 
444 public:

446  std::optional<VarInfo::ID> lookup(StringRef name) const;
447

449  /// of that name already exists: if `verifyUsage` is true, then will assert
450  /// that the variable has the same kind and a consistent location; otherwise,
451  /// when `verifyUsage` is false, this is a noop. Returns the identifier
