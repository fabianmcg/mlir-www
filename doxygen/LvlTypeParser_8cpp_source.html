19 // NOTE: There's no way for `FAILURE_IF_FAILED` to simultaneously support
20 // both `OptionalParseResult` and `InFlightDiagnostic` return types.
21 // We can get the compiler to accept the code if we returned "`{}`",
22 // however for `OptionalParseResult` that would become the nullopt result,
23 // whereas for `InFlightDiagnostic` it would become a result that can
24 // be implicitly converted to success. By using "`failure()`" we ensure
25 // that `OptionalParseResult` behaves as intended, however that means the
26 // macro cannot be used for `InFlightDiagnostic` since there's no implicit
27 // conversion.
28 #define FAILURE_IF_FAILED(STMT) \
29  if (failed(STMT)) { \
30  return failure(); \
31  }
32 
33 // Although `ERROR_IF` is phrased to return `InFlightDiagnostic`, that type
34 // can be implicitly converted to all four of `LogicalResult, `FailureOr`,
35 // `ParseResult`, and `OptionalParseResult`. (However, beware that the
36 // conversion to `OptionalParseResult` doesn't properly delegate to
37 // `InFlightDiagnostic::operator ParseResult`.)
38 //
39 // NOTE: this macro assumes `AsmParser parser` and `SMLoc loc` are in scope.
40 #define ERROR_IF(COND, MSG) \
41  if (COND) { \
42  return parser.emitError(loc, MSG); \
43  }

49 std::optional<DimLevelType> LvlTypeParser::lookup(StringRef str) const {
50  // NOTE: `StringMap::lookup` will return a default-constructed value if
51  // the key isn't found; which for enums means zero, and therefore makes
52  // it impossible to distinguish between actual zero-DimLevelType vs
53  // not-found. Whereas `StringMap::at` asserts that the key is found,
54  // which we don't want either.
55  const auto it = map.find(str);
56  return it == map.end() ? std::nullopt : std::make_optional(it->second);
57 }

69 ParseResult LvlTypeParser::parseLvlType(AsmParser &parser,
70  DimLevelType &out) const {
71  const auto loc = parser.getCurrentLocation();
72  StringRef strVal;
73  FAILURE_IF_FAILED(parser.parseOptionalKeyword(&strVal));
74  const auto lvlType = lookup(strVal);
75  ERROR_IF(!lvlType, "unknown level-type '" + strVal + "'")
76  out = *lvlType;
77  return success();
78 }
